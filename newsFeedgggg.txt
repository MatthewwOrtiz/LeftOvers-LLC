    var obj;
    var feedIdInOrder = new Array(5); // to store the Feed_ID in array. post0 (title0,content0,etc.) has feed id at feedIdInOrder[0]
    function update(){
      $.get("/data/newsfeed", function(result){
              obj = result;
    
              var idCount;
              var position = obj.length - 1;
             
              for ( idCount =0 ;idCount < 5 ; ++idCount){
                document.getElementById("title" + idCount).innerHTML = obj[position].Title;
                document.getElementById("date" + idCount).innerHTML = obj[position].DateTime;
                document.getElementById("content" + idCount).innerHTML = obj[position].Content;
                document.getElementById("thumbUp" + idCount).innerHTML = "  " + obj[position].ThumbsUp;
                document.getElementById("thumbDown" + idCount).innerHTML = "  " + obj[position].ThumbsDown;
      if(obj[position].Image){
                document.getElementById("newsImage" +idCount).src = obj[position].Image;
      }
                document.getElementById("comments" +idCount).innerHTML = "  " + obj[position].Comments;
                feedIdInOrder[idCount] = Number(obj[position].Feed_ID);       // getting feedIdInOrder for each element (integer value)
    
                --position;
              }        
        }); 
    }
